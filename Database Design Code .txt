데이터베이스 제작 //는 주석이니 mysql에 기입하지 말것.

create databases DBP; // DB 이름
use dbp;


create table userdata(
    userID varchar(20) primary key, // 기준값
    pwd varchar(20) not null,
    email_address varchar(20),
    phonenum char(11)
    );
// 가입한 멤버의 상세

create table group_name(
    group_name varchar(30) primary key, // 기준값
    member_id varchar(20) not null,
    project_name varchar(20) not null
    );
// 그룹에 대한 상세

create table project_data(
    project_name varchar(30) primary key,
    file_path varchar(400) not null,
    publisher_id varchar(20) not null,
    create_request_time timestamp not null,
    open_choice int not null
    );
// 프로젝트에 대한 상세


create table comment_data(
    project_name varchar(30) primary key,
    member_id varchar(20) not null,
    comment text not null,
    comment_upload_time timestamp not null
    );
// 코멘트에 대한 상세
